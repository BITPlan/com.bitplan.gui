41  public interface JsonAble extends AsJson, JsonValueMap {
42  
43    /**
44     * reinitialize me after being reloaded from json
45     */
46    public default void reinit() {
47      // override as you like
48    };
49  
50    /**
51     * save me to my Json File
52     * 
53     * @throws IOException
54     */
55    default void save() throws IOException {
56      File jsonFile = getJsonFile();
57      if (!jsonFile.getParentFile().isDirectory())
58        jsonFile.getParentFile().mkdirs();
59      FileUtils.writeStringToFile(jsonFile, this.asJson(), "UTF-8");
60    }
61  
62    default File getJsonFile() {
63      File jsonFile = getJsonFile(this.getClass().getSimpleName());
64      return jsonFile;
65    }
66  
67    /**
68     * get the json file for the given appName
69     * 
70     * @param appName
71     * @return the Json File
72     */
73    static File getJsonFile(String appName) {
74      String home = System.getProperty("user.home");
75      File configDirectory = new File(
76          home + "/." + Translator.APPLICATION_PREFIX + "/");
77      String jsonFileName = appName + ".json";
78      File jsonFile = new File(configDirectory, jsonFileName);
79      return jsonFile;
80    }
81  
82    /**
83     * get my values a a map
84     * 
85     * @return - the map of values
86     */
87    @SuppressWarnings({ "unchecked", "rawtypes" })
88    default Map<String, Object> asMap() {
89      Map<String, Object> map = new HashMap<String, Object>();
90      String json = this.asJson();
91      map = (Map<String, Object>) getGson().fromJson(json, map.getClass());
92      Field[] fields = this.getClass().getDeclaredFields();
93      for (Field field : fields) {
94        String typename = field.getType().getName();
95        // System.out.println(typename);
96        switch (typename) {
97        case "int":
98        case "java.lang.Integer":
99          Double di = (Double) map.get(field.getName());
100         if (di != null)
101           map.put(field.getName(), di.intValue());
102         break;
103       case "java.lang.Long":
104       case "long":
105         Double dl = (Double) map.get(field.getName());
106         if (dl != null)
107           map.put(field.getName(), dl.longValue());
108         break;
109       case "java.util.Date":
110         String dtext = (String) map.get(field.getName());
111         if (dtext != null) {
112           Date date;
113           try {
114             date = JsonManagerImpl.dateFormat.parse(dtext);
115             map.put(field.getName(), date);
116           } catch (ParseException e) {
117             // TODO proper handling
118             //if (debug)
119               // e.printStackTrace();
120           }
121         }
122         break;
123       default:
124         if (field.getType().isEnum()) {
125           String value = (String) map.get(field.getName());
126           if (value != null)
127             map.put(field.getName(),
128                 Enum.valueOf((Class<Enum>) field.getType(), value));
129         }
130       }
131     }
132 
133     /*
134      * Jackson version
135      * // https://stackoverflow.com/a/39544594/1497139
136      * ObjectMapper mapper = new ObjectMapper();
137      * mapper.setVisibility(PropertyAccessor.FIELD, Visibility.ANY);
138      * // Convert POJO to Map
139      * Map<String, Object> map = mapper.convertValue(this,
140      * new TypeReference<Map<String, Object>>() {
141      * });
142      */
143     return map;
144   }

146   /**
147    * initialize my values from the given map
148    * 
149    * @param map
150    */
151   @SuppressWarnings({ "rawtypes", "unchecked" })
152   public default void fromMap(Map<String, Object> map) {
153     Field[] fields = this.getClass().getDeclaredFields();
154     for (Field field : fields) {
155       String fieldName = field.getName();
156       field.setAccessible(true);
157       if (map.containsKey(fieldName)) {
158         try {
159           Object value=map.get(fieldName);
160           if (value instanceof String && field.getType().isEnum()) {
161             value=Enum.valueOf((Class<Enum>) field.getType(), (String)value);
162           }
163           field.set(this, value);
164         } catch (IllegalArgumentException | IllegalAccessException e) {
165           // TODO
166           // proper handling
167           // e.printStackTrace();
168         }
169       }
170     }
171   }
